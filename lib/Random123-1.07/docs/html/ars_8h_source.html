52 R123_STATIC_INLINE ars1xm128i_ctr_t ars1xm128i_R(unsigned int Nrounds, ars1xm128i_ctr_t in, ars1xm128i_key_t k){
53  __m128i kweyl = _mm_set_epi64x(R123_64BIT(0xBB67AE8584CAA73B), /* sqrt(3) - 1.0 */
54  R123_64BIT(0x9E3779B97F4A7C15)); /* golden ratio */
55  /* N.B. the aesenc instructions do the xor *after*
56  // so if we want to follow the AES pattern, we
57  // have to do the initial xor explicitly */
58  __m128i kk = k.v[0].m;
59  __m128i v = _mm_xor_si128(in.v[0].m, kk);
60  ars1xm128i_ctr_t ret;
61  R123_ASSERT(Nrounds<=10);
62  if( Nrounds>1 ){
63  kk = _mm_add_epi64(kk, kweyl);
64  v = _mm_aesenc_si128(v, kk);
65  }
66  if( Nrounds>2 ){
67  kk = _mm_add_epi64(kk, kweyl);
68  v = _mm_aesenc_si128(v, kk);
69  }
70  if( Nrounds>3 ){
71  kk = _mm_add_epi64(kk, kweyl);
72  v = _mm_aesenc_si128(v, kk);
73  }
74  if( Nrounds>4 ){
75  kk = _mm_add_epi64(kk, kweyl);
76  v = _mm_aesenc_si128(v, kk);
77  }
78  if( Nrounds>5 ){
79  kk = _mm_add_epi64(kk, kweyl);
80  v = _mm_aesenc_si128(v, kk);
81  }
82  if( Nrounds>6 ){
83  kk = _mm_add_epi64(kk, kweyl);
84  v = _mm_aesenc_si128(v, kk);
85  }
86  if( Nrounds>7 ){
87  kk = _mm_add_epi64(kk, kweyl);
88  v = _mm_aesenc_si128(v, kk);
89  }
90  if( Nrounds>8 ){
91  kk = _mm_add_epi64(kk, kweyl);
92  v = _mm_aesenc_si128(v, kk);
93  }
94  if( Nrounds>9 ){
95  kk = _mm_add_epi64(kk, kweyl);
96  v = _mm_aesenc_si128(v, kk);
97  }
98  kk = _mm_add_epi64(kk, kweyl);
99  v = _mm_aesenclast_si128(v, kk);
100  ret.v[0].m = v;
101  return ret;
102 }

121  ars1xm128i_ctr_t c128;
122  ars1xm128i_key_t k128;
123  c128.v[0].m = _mm_set_epi32(c.v[3], c.v[2], c.v[1], c.v[0]);
124  k128.v[0].m = _mm_set_epi32(k.v[3], k.v[2], k.v[1], k.v[0]);
125  c128 = ars1xm128i_R(Nrounds, c128, k128);
126  _mm_storeu_si128((__m128i*)&c.v[0], c128.v[0].m);
